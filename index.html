init([]) ->
    case application:get_env(riak_jmx, enabled) of
        {ok, true} ->
            %% Trap exits so that we get a chance 
            %% to stop the JMX process
            process_flag(trap_exit, true),
            Port = jmx(),
            {ok, #state{ retry = 0, port = Port }};
        _ ->
            ignore
    end.
